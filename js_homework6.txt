function findValueByKey(companyName) {
    let client = company.clients.find(client => client.name === companyName);

    if (client) {
        return client;
    }

    for (let i = 0; i < company.clients.length; i++) {
        let partners = company.clients[i].partners;
        
        if (partners) {
            client = partners.find(partner => partner.name === companyName);
            if (client) {
                return client;
            }
            
            for (let j = 0; j < partners.length; j++) {
                let innerPartners = partners[j].partners;
                if (innerPartners) {
                    client = innerPartners.find(partner => partner.name === companyName);
                    if (client) {
                        return client;
                    }
                }
            }
        }
    }

    return null;
}
